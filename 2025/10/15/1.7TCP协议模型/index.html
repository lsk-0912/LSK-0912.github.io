TCP协议模型

【一】一代

  • client.py
```python from conf import setting import socket # 创建一个server对象：server:服务器 # AF_INET:当前连接是基于网络的套接字 # SOCK_STREAM:连接模式是TCP协议的流式模式 # SOCK_STREAM表示创建一个面向连接的、基于TCP协议的socket类型。它是一种可靠的、双向的、点对点的通信协议，提供面向连接的、可靠的数据传输服务。 client = socket.socket(family=socket.AF_INET, type=socket.SOCK_STREAM) # 绑定IP和PORT # connect:连接 client.connect(setting.ADDR) # 直接发送数据 to_server_send_data = f"这是来自客户端的一条数据！" #转码 to_server_send_data = to_server_send_data.encode() #发送 #send：将数据to_server_send_data传给客户端 # 当你发送的数据较大时，可以做一个循环，分批发送 client.send(to_server_send_data) #接收到服务端回的消息 from_server_recv_data = client.recv(1024) #接收到信息记得解码 from_server_recv_data = from_server_recv_data.decode() print(from_server_recv_data) #关闭连接对象 client.close()

【二】二代—空值的时候

  • client.py
```python from conf import setting import socket client = socket.socket(family=socket.AF_INET, type=socket.SOCK_STREAM) client.connect(setting.ADDR) while True: to_server_send_data = input("回应服务端：=--").strip() if not to_server_send_data: print("数据不能为空！") continue if to_server_send_data == "q": print("当前连接已退出！") break to_server_send_data = to_server_send_data.encode() client.send(to_server_send_data) from_server_recv_data = client.recv(1024) from_server_recv_data = from_server_recv_data.decode() print(f"接收服务端的信息--{from_server_recv_data}") #关闭连接对象 client.close()
